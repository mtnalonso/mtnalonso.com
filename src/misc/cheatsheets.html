Tools

nmap

# ICMP (Ping) scan
nmap -sP <target-ip-address>

# Ping sweep
nmap -sP 192.168.0.1-255

# TCP scan
nmap -sT <target-ip-address>

# Stealth (SYN) scan
nmap -sS <target-ip-address>

# XMAS scan
nmap -sX <target-ip-address>

# ACK scan
nmap -sA <target-ip-address>

# WINDOW scan
nmap -sW <target-ip-address>

# UDP scan
nmap -sU <target-ip-address>
